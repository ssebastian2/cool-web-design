const facts = [
  "i really like making stories, play video games and training.",
  "My favorite sport is badtminton or martial arts.",
  "I enjoy playing games on weekends.",
  "I like train or make stories on weekends."
];

const factParagraph = document.getElementById("fact");
const button = document.getElementById("changeButton");

button.addEventListener("click", () => {
  const randomIndex = Math.floor(Math.random() * facts.length);
  factParagraph.textContent = facts[randomIndex];
});
